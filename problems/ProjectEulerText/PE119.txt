import  math

dps_sequence = []

for a in range(2, math.floor(math.sqrt(700000))):
    for b in range(2, math.floor(math.log(math.sqrt(700000), 2)) + 1):
            number = a ** b
            if sum(int(str(number)[i]) for i in range(len(str(number)))) == a:
                dps_sequence.append(number)
dps_sequence.sort()

print(dps_sequence[29])
