def sum_of_divisors(k):
    sum = 0
    for i in range(1, k):
        if k % i == 0:
            sum += i
    return sum


dict_of_div = {}
for j in range(1, 10000):
    dict_of_div[j] = sum_of_divisors(j)

print(dict_of_div)

amicable_sum = 0
for i in dict_of_div:
    try:
        if dict_of_div[dict_of_div[i]] == i and dict_of_div[i] != i:
            amicable_sum += i
            print(i)
    except KeyError:
        pass



print(amicable_sum)
