import math
Palindromes = []

for a in range(100, 999):
    for b in range(100, 999):
        New = str(a*b)
        NewScore = 0
        for x in range(0, math.floor(len(New)/2)):
            if int(New[x]) == int(New[len(New) - x - 1]):
                NewScore += 1
        if NewScore == math.floor(len(New)//2):
            Palindromes.append(int(New))
print(Palindromes)
print(max(Palindromes))
